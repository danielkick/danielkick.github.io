\[E_{ion} = \frac{RT}{zF}ln(\frac{ion_{outside}}{ion_{inside}})\] \[E_{ion} = \frac{61.5}{z}log_{10}(\frac{ion_{outside}}{ion_{inside}})\]

nernst_stp <- function(z = 1,
                       ion_o = 1,
                       ion_i = 10){
  E_ion <- (61.5/z)*log10(ion_o/ion_i)
  return(E_ion)
}
